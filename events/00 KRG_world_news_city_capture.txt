###########################
# City Capture Events
###########################
#
# If you want to add more city capture events:
#
# First add an entry for the city in KR_on_actions_city_capture in the common\on_actions folder
#
# Then add a news event here - you can copy worldnewscitycapture.251 for the most basic version
# which provides a separate comment for the owner (old_controller), the conqueror (new_controller), and then everyone else
#

add_namespace = worldnewscitycapture

# The Fall of London
news_event = {
	id = worldnewscitycapture.251
	immediate = { log = "[GetDateText]: [Root.GetName]: event worldnewscitycapture.251 FROM = [From.GetName]" }
	title = worldnewscitycapture.251.t
	desc = {
		text = worldnewscitycapture.251.d.generic
		trigger = {
			event_target:new_controller = {
				NOT = {
					tag = UOB
					tag = MXF
				}
			}
		}
	}
	desc = {
		text = worldnewscitycapture.251.d.uob
		trigger = {
			event_target:new_controller = {
				OR = {
					tag = UOB
					tag = MXF
				}
			}
		}
	}
	picture = GFX_news_event_capture_london

	major = yes
	is_triggered_only = yes

	trigger = {
		wants_world_news = yes
	}

	immediate = {
		126 = {
			set_state_flag = state_has_fallen
			owner = { save_event_target_as = old_controller }
			controller = { save_event_target_as = new_controller }
		}
	}

	option = { #This is only a temporary setback
		name = worldnewscitycapture.251.a
		trigger = {
			tag = event_target:old_controller
			event_target:new_controller = {
				NOT = {
					tag = UOB
					tag = MXF
				}
			}
		}
	}

	option = { #This is the beginning of the end
		name = worldnewscitycapture.251.b
		trigger = {
			tag = event_target:new_controller
			NOT = {
				tag = UOB
				tag = MXF
			}
		}
	}

	option = { #The Sun sets on the British
		name = worldnewscitycapture.251.c
		trigger = {
			NOT = {
				tag = event_target:old_controller
				tag = event_target:new_controller
				event_target:new_controller = {
					OR = {
						tag = UOB
						tag = MXF
					}
				}
			}
		}
	}
	
	option = { #A nightmare reborn.
		name = worldnewscitycapture.251.a.uob
		trigger = {
			OR ={
				tag = event_target:old_controller
				tag = ENT
				tag = GER
			}
			event_target:new_controller = {
				OR = {
					tag = UOB
					tag = MXF
				}
			}
		}
	}
	
	option = { #We shall keep the red flag flying here
		name = worldnewscitycapture.251.b.uob
		trigger = {
			tag = event_target:new_controller
			event_target:new_controller = {
				OR = {
					tag = UOB
					tag = MXF
				}
			}
		}
	}
	
	option = { #History repeats itself.
		name = worldnewscitycapture.251.c.uob
		trigger = {
			NOT = {
				tag = event_target:old_controller
				tag = event_target:new_controller

				tag = GER
