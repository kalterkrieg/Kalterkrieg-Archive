	#Revolutionary Victory
	id = krg_mcw.3
	title = krg_mcw.3.t
	desc = krg_mcw.3.d
	immediate = {
		log = "[GetDateText]: [Root.GetName]: krg_mcw.3"
	}
	trigger = {
		tag = AMR
		has_war_with = MOR
		is_valid_peace_target = yes
		MOR = {
			surrender_progress > 0.6
		}
		NOT = {
			has_global_flag = moroccan_revolution_ended
		}
	}
	fire_only_once = yes
	option = {
		#Annexes MOR
		name = krg_mcw.3.a
		hidden_effect = {
			country_event = {
				id = krg_mcw.6
				days = 365
			}
			MOR = {
				every_core_state = {
					add_core_of = AMR
				}
			}
		}
		annex_country = {
			target = MOR
			transfer_troops = no
		}
		CAN = {
			add_to_faction = AMR
			if = {
				limit = {
					SPR = {
						NOT = { has_government = radical_socialist }
					}
				}
				add_to_faction = SPR
			}
		}
		AMR = {
			set_temp_variable = {
				temp_level = 4
			}
			set_accord_alignment = yes
		}
		remove_from_array = {
			global.influence_targets = 27
		}
		add_to_array = {
			global.influence_log = 30
		}
		add_to_array = {
			global.influence_log_val = 15
		}
		influence_recalc = yes
		news_event = {
			id = worldnews.406
		}
		set_global_flag = moroccan_revolution_ended
	}
}
