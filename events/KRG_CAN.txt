				}

				}
				30 = {

				}
			}
			random_list = {
				30 = {
					add_to_variable = { opponent_party_psuedo_pp_arr^2 = 10 }
				}
				40 = {
					add_to_variable = { opponent_party_psuedo_pp_arr^2 = 15 }
				}
				30 = {
					add_to_variable = { opponent_party_psuedo_pp_arr^2 = 20 }
				}
			}
			random_list = {
				30 = {
					add_to_variable = { opponent_party_psuedo_pp_arr^3 = 10 }
				}
				40 = {
					add_to_variable = { opponent_party_psuedo_pp_arr^3 = 15 }
				}
				30 = {
					add_to_variable = { opponent_party_psuedo_pp_arr^3 = 20 }
				}
			}

				limit = { NOT = { check_variable = { CAN_election_party_support = 0 } } }
				set_variable = { party_decide_index = 0 }
				CAN_election_decide_campaign = yes
			}
			if = {
				limit = { NOT = { check_variable = { CAN_election_party_support = 1 } } }
				set_variable = { party_decide_index = 1 }
				CAN_election_decide_campaign = yes

				limit = { NOT = { check_variable = { CAN_election_party_support = 2 } } }

				CAN_election_decide_campaign = yes

			if = {
				limit = { 
					NOT = { 
						check_variable = { CAN_election_party_support = 3 }

					}

				set_variable = { party_decide_index = 3 }

			}
		}
	}

		name = krg_can_elect.1.a
