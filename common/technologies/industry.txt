							}
						}
						if = {
							limit = {
								NOT = {
									has_state_flag = steelmaking_1
								}
							}
							hidden_effect = {
								set_temp_variable = { state_steel_amount = THIS.resource@steel }
								multiply_temp_variable = { state_steel_amount = 0.05 } #adding 5%
								# round_temp_variable = state_steel_amount
								add_resource = {
									type = steel
									amount = state_steel_amount
								}
							}
							custom_effect_tooltip = add_5_percent_steel
							log = "[THIS.GetName] Added 5% of steel"
							set_state_flag = steelmaking_1
						}
						else = {
							custom_effect_tooltip = steelmaking_1_finished
						}
					}
				}
