		every_owned_state = {
			# senate calculations
			if = {
				limit = {
					check_variable = { election_count < 2 }
				}
				find_highest_in_array = { array = electoral_pie_chart_number_array index = pie_highest_index }
				set_temp_variable = { temp = pie_highest_index }
				add_to_temp_variable = { temp = 1 }
				if = {
					limit = {
						check_variable = { election_count = 1 }
					}
					set_variable = { senate_seat_party_2 = temp }
				}
				else = {
					set_variable = { senate_seat_party_1 = temp }
				}
				add_to_variable = { election_count = 1 }
			}
			else = {
				clear_variable = election_count
			}
			# house calculations
			clear_variable = temp_total
			log = "[THIS.GetName]"
			for_each_loop = {
				array = electoral_pie_chart_number_array
				set_temp_variable = { temp_seats = electoral_pie_chart_number_array^i } # put a party popularity to a temp var
				divide_temp_variable = { temp_seats = 100 } # divide party popularity so you get 0.65 from 65% for example
				multiply_temp_variable = { temp_seats = house_seats } # multiply that by number of seats to get seat share
				round_temp_variable = temp_seats # round it up
				set_variable = { electoral_seat_number_array^i = temp_seats } # set electoral seat to new value
				add_to_variable = { temp_total = temp_seats }
				log = "[?i] - [?electoral_seat_number_array^i] ([?temp_total])"
			}
			# extra to make sure it adds up to electoral seats in state
			if = {
				limit = {
					check_variable = { temp_total < house_seats } # if total is less than number of seats
				}
				set_temp_variable = { temp = house_seats } # put number of seats to a temp var
				subtract_from_temp_variable = { temp = temp_total } # get seat difference
				find_highest_in_array = { array = electoral_seat_number_array index = highest_index }
				add_to_variable = { electoral_seat_number_array^highest_index = temp } # add seat difference to most popular party
			}
			else_if = {
				limit = {
					check_variable = { temp_total > house_seats } # if total is more than number of seats
				}
				subtract_from_variable = { temp_total = house_seats } # get seat difference
				find_highest_in_array = { array = electoral_seat_number_array index = highest_index }
				subtract_from_variable = { electoral_seat_number_array^highest_index = temp_total } # subtract seat difference from most popular party
			}
		}
	}
}
NEE_calculate_president = {
	hidden_effect = {
		every_owned_state = {
			limit = {
				is_core_of = NEE
			}
			find_highest_in_array = { 
				array = electoral_pie_chart_number_array 
				index = index 
			}
			if = {
				limit = {
					check_variable = { index = 0 }
				}
				add_to_variable = { progressives_electoral_votes = electoral_vote }
				set_variable = {
					electoral_lean = 1
				}
			}
			else_if = {
				limit = {
					check_variable = { index = 1 }
				}
				add_to_variable = { soclib_electoral_votes = electoral_vote }
				set_variable = {
					electoral_lean = 2
				}
			}
			else_if = {
				limit = {
					check_variable = { index = 2 }
				}
				add_to_variable = { marlib_electoral_votes = electoral_vote }
				set_variable = {
					electoral_lean = 3
				}
			}
			else_if = {
				limit = {
					check_variable = { index = 3 }
				}
				add_to_variable = { soccon_electoral_votes = electoral_vote }
				set_variable = {
					electoral_lean = 4
				}
			}
			if = { #Before 1952 progressives vote for the democrats for president
				limit = { date < 1952.1.1 }
				add_to_variable = {
					var = NEE.soccon_electoral_votes
					value = progressives_electoral_votes
				}
			}
			else = {
				add_to_variable = { #after 1942 progressives vote for soc lib dems for president
					var = NEE.soclib_electoral_votes
					value = progressives_electoral_votes
				}
			}
			if = {
				limit = { date < 1952.1.1 } 
				add_to_variable = {
					var = NEE.soccon_electoral_votes
					value = soclib_electoral_votes
				}
			}
			else = {
				add_to_variable = { #After 1952 Social liberals vote for soc lib democrats
					var = NEE.soclib_electoral_votes
					value = soclib_electoral_votes
				}
			}

					var = NEE.soccon_electoral_votes
					value = soccon_electoral_votes

			}

				}
				add_to_variable = {
					var = NEE.soclib_electoral_votes
					value = soccon_electoral_votes
				}
			}
			else = {

					var = NEE.marlib_electoral_votes
					value = soccon_electoral_votes
				}
			}

	}

NEE_create_pie_chart = {
	clear_array = pie_chart
	clear_array = temp_pie
	for_each_loop = {
		array = electoral_pie_chart_number_array
		add_to_array = { temp_pie = electoral_pie_chart_number_array^i }
	}
	set_temp_variable = { temp = 0 }
	set_temp_variable = { temp_index = 1 }
	while_loop_effect = {
		limit = {
			check_variable = { temp < temp_pie^num }
		}
		add_to_variable = { temp_pie^temp_index = temp_pie^temp }
		add_to_temp_variable = { temp = 1 }
		add_to_temp_variable = { temp_index = 1 }
	}
	set_temp_variable = { temp = 0 }
	while_loop_effect = {
		limit = {
			check_variable = { temp < temp_pie^num }
		}
		resize_array = { array = pie_chart value = electoral_seat_id_array^temp size = temp_pie^temp }
		add_to_temp_variable = { temp = 1 }
	}
}

	set_temp_variable = { v =  pop_type_added }

	set_temp_variable = { temp =  electoral_pie_chart_number_array^pop_type_added }
	add_to_temp_variable = { temp = pop_value_added }
	if = {
		limit = {
			check_variable = { temp > 100 }
		}
		subtract_from_temp_variable = { temp = 100 }
		subtract_from_temp_variable = { pop_value_added = temp }
	}
	for_each_loop = {
		array = electoral_pie_chart_number_array
		add_to_array = { percentage_popularity = electoral_pie_chart_number_array^i }

		set_variable = { electoral_pie_chart_number_array^i = percentage_popularity^i }
	}
	clear_array = percentage_popularity
	TEX_create_pie_chart = yes
}
