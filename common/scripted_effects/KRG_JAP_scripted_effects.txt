# spits out value for the power bar progress of each faction based on influence
JAP_calculate_influence = {
	clear_array = power_brokers_influence_array
	clear_array = power_brokers_influence_percent_array
	add_to_array = { power_brokers_influence_array = 0 }
	add_to_array = { power_brokers_influence_array = 0 }
	add_to_array = { power_brokers_influence_array = 0 }

	for_each_scope_loop = {
		array = power_brokers_array

		for_each_loop = {
			array = power_brokers_influence_array

			add_to_variable = { JAP.power_brokers_influence_array^i = power_brokers_influence_array^i }
		}
		log = "[?power_brokers_influence_array^i]"
	}

	for_each_loop = {
		array = power_brokers_influence_array

		divide_variable = { power_brokers_influence_array^i = power_brokers_array^num }
		round_variable = power_brokers_influence_array^i
		add_to_variable = { JAP.temp_total = power_brokers_influence_array^i }
	}

	# extra to make sure it adds up to 100
	if = {
		limit = {
			check_variable = { temp_total < 100 }
		}
		set_temp_variable = { temp = 100 }
		subtract_from_temp_variable = { temp = temp_total }
		find_highest_in_array = { array = power_brokers_influence_array index = highest_index }
		add_to_variable = { power_brokers_influence_array^highest_index = temp }
	}
	else_if = {
		limit = {
			check_variable = { temp_total > 100 }
		}
		subtract_from_temp_variable = { temp_total = 100 }
		find_highest_in_array = { array = power_brokers_influence_array index = highest_index }
		subtract_from_variable = { power_brokers_influence_array^highest_index = temp_total }
	}

	for_each_loop = {
		array = power_brokers_influence_array

		add_to_array = { power_brokers_influence_percent_array = power_brokers_influence_array^i }
	}

	add_to_variable = { power_brokers_influence_percent_array^1 = power_brokers_influence_percent_array^0 }
	set_variable = { power_brokers_influence_percent_array^2 = 100 }
}

# general parliament code stuff yada yada
JAP_refresh_parliament = {
	clear_array = house_of_representatives_main_array
	clear_array = house_of_representatives_seat_array
	clear_array = temp_seat

	add_to_array = { house_of_representatives_seat_array = 212 }
	add_to_array = { house_of_representatives_seat_array = 172 }
	add_to_array = { house_of_representatives_seat_array = 81 }

	set_variable = { house_of_representatives_support_percent = house_of_representatives_support }
	divide_variable = { house_of_representatives_support_percent = house_of_representatives_seats }
	multiply_variable = { house_of_representatives_support_percent = 100 }
	round_variable = house_of_representatives_support_percent

	for_each_loop = {
		array = house_of_representatives_seat_array

		add_to_array = { temp_seat = house_of_representatives_seat_array^i }
	}

	set_temp_variable = { temp = 0 }
	set_temp_variable = { temp_index = 1 }
	set_temp_variable = { temp_size = temp_seat^num }

	while_loop_effect = {
		limit = {
			check_variable = { temp < temp_size }
		}
		add_to_variable = { temp_seat^temp_index = temp_seat^temp }
		add_to_temp_variable = { temp = 1 }
		add_to_temp_variable = { temp_index = 1 }
	}

	set_temp_variable = { temp = 0 }

	while_loop_effect = {
		limit = {
			check_variable = { temp < temp_size }
		}
		resize_array = { array = house_of_representatives_main_array value = house_of_representatives_member_array^temp size = temp_seat^temp }
		add_to_temp_variable = { temp = 1 }
	}

	add_to_variable = { show_JAP_parliament = 1 }
}

JAP_build_parliament_member_seat = {
	set_temp_variable = { temp_seat_count = house_of_representatives_seat_array^temp_index }

	while_loop_effect = {
		limit = {
			check_variable = { temp_seat_count > 0 }
		}
		add_to_array = { house_of_representatives_main_array = temp_current_member }
		subtract_from_temp_variable = { temp_seat_count = 1 }
	}
}
