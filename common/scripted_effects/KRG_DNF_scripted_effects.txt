########################################################
# Update seats in parliament based on party popularity
# Usage:
########################################################
# DNF_chamber_of_deputies_election_effect = yes
########################################################
DNF_chamber_of_deputies_election_effect = {
	add_to_temp_array = { temp_popularity_array = party_popularity@authoritarian_socialist }
	add_to_temp_array = { temp_popularity_array = party_popularity@radical_socialist }
	add_to_temp_array = { temp_popularity_array = party_popularity@social_democrat }
	add_to_temp_array = { temp_popularity_array = party_popularity@revolutionary_republican }
	add_to_temp_array = { temp_popularity_array = party_popularity@national_populist }
	add_to_temp_array = { temp_popularity_array = party_popularity@market_liberal }
	add_to_temp_array = { temp_popularity_array = party_popularity@social_liberal }
	add_to_temp_array = { temp_popularity_array = party_popularity@social_conservative }
	add_to_temp_array = { temp_popularity_array = party_popularity@authoritarian_democrat }
	add_to_temp_array = { temp_popularity_array = party_popularity@paternal_autocrat }

	for_each_loop = {
		array = temp_popularity_array

		multiply_temp_variable = { temp_popularity_array^i = 732 } # get number of seats for each party via multiplying popularity percent with total seats
		round_temp_variable = temp_popularity_array^i
		add_to_temp_variable = { temp_total = temp_popularity_array^i }
	}

	# sanity check to ensure total adds up to 732
	if = {
		limit = {
			check_variable = { temp_total < 732 }
		}
		set_temp_variable = { temp_seat = 732 }
		subtract_from_temp_variable = { temp_seat = temp_total }
		find_highest_in_array = { array = temp_popularity_array index = highest_index }
		add_to_temp_variable = { temp_popularity_array^highest_index = temp_seat }
	}
	else_if = {
		limit = {
			check_variable = { temp_total > 732 }
		}
		subtract_from_temp_variable = { temp_total = 732 }
		find_highest_in_array = { array = temp_popularity_array index = highest_index }
		subtract_from_temp_variable = { temp_popularity_array^highest_index = temp_total }
	}

	# set seats for each party
	for_each_loop = {
		array = DNF_chamber_of_deputies_party_array

		set_variable = { v:DNF_chamber_of_deputies_seat = temp_popularity_array^i }
	}

	DNF_chamber_of_deputies_reload = yes
}

		DNF_chamber_of_deputies_reload = yes
	}
}

# Maho's parliament builder version 5
DNF_chamber_of_deputies_reload = {
	clear_array = DNF_chamber_of_deputies_main_array
	clear_array = DNF_chamber_of_deputies_cosmetic_party_array
	clear_array = DNF_chamber_of_deputies_coalition_array
	clear_array = DNF_chamber_of_deputies_opposition_array
	clear_variable = DNF_chamber_of_deputies_coalition_seat
	clear_variable = DNF_chamber_of_deputies_opposition_seat

	# Cut parties with no seats from gui
	for_each_loop = {
		array = DNF_chamber_of_deputies_party_array

		if = {
			limit = {
				check_variable = { v:DNF_chamber_of_deputies_seat > 0 }
			}
			add_to_array = { DNF_chamber_of_deputies_cosmetic_party_array = v }
		}
	}

	# Add party to respective alignment arrays
	for_each_loop = {
		array = DNF_chamber_of_deputies_cosmetic_party_array

		if = {
			limit = {
				check_variable = { v:DNF_chamber_of_deputies_alignment = 1 }
			}
			add_to_array = { DNF_chamber_of_deputies_coalition_array = v }
			add_to_variable = { DNF_chamber_of_deputies_coalition_seat = v:DNF_chamber_of_deputies_seat }
		}
		else = {
			add_to_array = { DNF_chamber_of_deputies_opposition_array = v }
			add_to_variable = { DNF_chamber_of_deputies_opposition_seat = v:DNF_chamber_of_deputies_seat }
		}
	}

	# Resort parties split into two groups
	clear_array = DNF_chamber_of_deputies_cosmetic_party_array

	for_each_loop = {
		array = DNF_chamber_of_deputies_coalition_array

		add_to_array = { DNF_chamber_of_deputies_cosmetic_party_array = v }
	}

	for_each_loop = {
		array = DNF_chamber_of_deputies_opposition_array

		add_to_array = { DNF_chamber_of_deputies_cosmetic_party_array = v }
	}

	# Add to main array each party's seats
	for_each_loop = {
		array = DNF_chamber_of_deputies_cosmetic_party_array

		set_temp_variable = { global.seat_id = v }

		var:v = {
			if = {
				limit = {
					check_variable = { DNF_chamber_of_deputies_seat > 0 }
				}
				for_loop_effect = {
					start = 1
					end = DNF_chamber_of_deputies_seat
					compare = less_than_or_equals

					add_to_array = { DNF.DNF_chamber_of_deputies_main_array = global.seat_id }
				}
			}
		}
	}

	DNF_chamber_of_deputies_semicircle = yes

	for_each_loop = {
		array = DNF_chamber_of_deputies_x_array

		if = {
			limit = {
				check_variable = { i < DNF_chamber_of_deputies_coalition_seat }
			}
			add_to_variable = { DNF_chamber_of_deputies_x_array^i = -5 }
		}
		else = {
			add_to_variable = { DNF_chamber_of_deputies_x_array^i = 5 }
		}
	}

	DNF_chamber_of_deputies_screen_refresh = yes
}

DNF_chamber_of_deputies_screen_refresh = {
	add_to_variable = { show_DNF_chamber_of_deputies = 1 }
}

	}
}
