##how to define a law effect

##CAN_law_[law id]_effect = {
##	CAN_law_effect = yes ##you must have CAN_law_effect = yes or your clauses wont work
##	any other effects, so adding a dynamic modifer or smth
##}

##how to define a law fail effect
##CAN_law_[law id]_fail = {
##	any effects, pp hit, event or other
##}


##how to define a clause effect
##CAN_clause_[clause id]_effect = {
##	any effects, like adding to a variable for a dynamic modifier
##}

##how to define a clause add effect
##CAN_clause_[clause id]_add_effect = {
##	any effects, like adding to a variable for a dynamic modifier
##}

##how to define a clause remove effect
##CAN_clause_[clause id]_remove_effect = {
##	any effects, like adding to a variable for a dynamic modifier
##}

##how to define a law effect for the AI -- this system is NOT coded for the ai, it will not work for them if you dont do this
##CAN_law_[law id]_ai_setup = {
##	CAN_law_effect = yes ##you must have CAN_law_effect = yes or your clauses wont work
##	you should set up clauses here, so something like this to add clauses 1, 4, and 3:
##  CAN_clause_1_add_effect = yes
##	CAN_clause_4_add_effect = yes
##	CAN_clause_3_add_effect = yes
##	that is all you have to do for the ai, the rest of the code will handle it
##}
