IRE_upgrade_template_and_divisions = { #this is meant to upgrade the militia division linearly, create the division if it hasn't been created already. 
	if = {
		limit = {
			has_variable = IRE_militia_template_level
			check_variable =  { IRE_militia_template_level < 2 }
		}
		add_units_to_division_template = {
			template_name = "An Fórsa Cosanta Áitiúil"
			regiments = {
				militia = 0
				militia = 0 
			}
		}
	}
	else_if = {
		limit = {
			has_variable = IRE_militia_template_level
			check_variable =  { IRE_militia_template_level < 3 }
		}
		add_units_to_division_template = {
			template_name = "An Fórsa Cosanta Áitiúil"
			regiments = {
				militia = 1
				militia = 1 
			}
		}
	}
	else_if = {
		limit = {
			has_variable = IRE_militia_template_level
			check_variable =  { IRE_militia_template_level < 4 }

		}
		add_units_to_division_template = {
			template_name = "An Fórsa Cosanta Áitiúil"
			regiments = {
				militia = 1
			}
			support = {
				engineer = 0
			} 
		}
	}
	if = { #Increasing the Militia Level variable 
		limit = {
			has_variable = IRE_militia_template_level
			check_variable = { IRE_militia_template_level < 10 }
		}
		add_to_variable = { IRE_militia_template_level = 1 }
	}

	else_if = {
		limit = {
			NOT = { has_variable = IRE_militia_template_level }  
		}
		set_variable = { IRE_militia_template_level = 1 }
	}
}

IRE_calculate_nationalist_status = { 
	if = { #0 Calm
		limit = { 
			check_variable = { IRE_nationalist_opinion > 19 }
			check_variable = { IRE_nationalist_opinion < 50 }
			check_variable = { IRE_unionist_strength < 50 }
	 	}
		set_variable = { IRE_nationalist_status = 0 }
	}

	else_if = { #1 tense
		limit = {
			OR = {
				AND = {
					check_variable = { IRE_nationalist_opinion > 0 }
					check_variable = { IRE_nationalist_opinion < 20 }
					check_variable = { IRE_unionist_strength < 65 }
				}
				AND = {
					check_variable = { IRE_nationalist_opinion > 19 }
					check_variable = { IRE_nationalist_opinion < 50 }
					check_variable = { IRE_unionist_strength > 49 }
				}
			}
			
	 	}
		set_variable = { IRE_nationalist_status = 1 }
	}

	else_if = { #2 agitated
		limit = {
			OR = {
				AND = {
					check_variable = { IRE_nationalist_opinion > -19 }
					check_variable = { IRE_nationalist_opinion < 0 }
					check_variable = { IRE_unionist_strength < 80 }
				}
				AND = {
					check_variable = { IRE_nationalist_opinion > 0 }
					check_variable = { IRE_nationalist_opinion < 20 }
					check_variable = { IRE_unionist_strength > 64 }
				}
			}
	 	}
		set_variable = { IRE_nationalist_status = 2 }
	}

	else_if = { #3 angry
		limit = {
			OR = {
				AND = {
					check_variable = { IRE_nationalist_opinion > -35 }
					check_variable = { IRE_nationalist_opinion < -20 }
				}
				AND = {
					check_variable = { IRE_nationalist_opinion > 0 }
					check_variable = { IRE_nationalist_opinion < 20 }
					check_variable = { IRE_unionist_strength > 79 }
				}
			}
				
	 	}
		set_variable = { IRE_nationalist_status = 3 }
	}

	else_if = { #4 hostile
		limit = { 
			check_variable = { IRE_nationalist_opinion < -34 }
			check_variable = { IRE_nationalist_strength  < 59 }

			check_variable = { IRE_unionist_opinion < -34 }
			check_variable = { IRE_unionist_strength < 59 }
	 	}
		set_variable = { IRE_nationalist_status = 4 }
	}

	else_if = { #5 inconflict
		limit = { 
			check_variable = { IRE_nationalist_opinion < -34 }
			check_variable = { IRE_nationalist_strength  > 59 }

			check_variable = { IRE_unionist_opinion < -34 }
			check_variable = { IRE_unionist_strength > 59 }
	 	}
		set_variable = { IRE_nationalist_status = 5 }
	}

}

	set_temp_variable = { IRE_temp_election_effect_meter = 0 }

	add_to_temp_variable = { IRE_temp_election_effect_meter = IRE_nationalist_status }
	multiply_temp_variable = { IRE_temp_election_effect_meter = 2 }
	add_to_temp_variable = { IRE_temp_election_effect_meter = IRE_unionist_status }

	log = "[GetDateText]: [Root.GetName]: IRE_temp_election_effect_meter: [?IRE_temp_election_effect_meter]"
	if = {
		limit = { 
			NOT = {
				has_country_flag = IRE_1952_election_done
			}
		}
		if = {
			limit = { 
				check_variable = { IRE_temp_election_effect_meter > -1 }
				check_variable = { IRE_temp_election_effect_meter < 4 }
			}
			country_event = { id = krg_ire_elections.31 }
		}
	
		else_if = {
			limit = { 
				check_variable = { IRE_temp_election_effect_meter > 3 }
				check_variable = { IRE_temp_election_effect_meter < 6 }
			}
			country_event = { id = krg_ire_elections.32 }

			add_to_variable = { IRE_1952_election_shift = 1 }
		}
	
		else_if = {
			limit = { 
				check_variable = { IRE_temp_election_effect_meter > 5 }
				check_variable = { IRE_temp_election_effect_meter < 10 }
			}
			country_event = { id = krg_ire_elections.33 }

			add_to_variable = { IRE_1952_election_shift = 2 }
		}
	
		else_if = {
			limit = { 
				check_variable = { IRE_temp_election_effect_meter > 9 }
			}
			country_event = { id = krg_ire_elections.34 }
