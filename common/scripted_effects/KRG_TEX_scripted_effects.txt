TEX_calculate_electoral_seat = {
	TEX = {
		every_owned_state = {
			# senate calculations
			if = {
				limit = {
					check_variable = {
						election_count < 2
					}
				}
				find_highest_in_array = {
					array = electoral_pie_chart_number_array
					index = pie_highest_index
				}
				set_temp_variable = {
					temp = pie_highest_index
				}
				add_to_temp_variable = {
					temp = 1
				}
				if = {
					limit = {
						check_variable = {
							election_count = 1
						}
					}
					set_variable = {
						senate_seat_party_2 = temp
					}
				}
				else = {
					set_variable = {
						senate_seat_party_1 = temp
					}
				}
				add_to_variable = {
					election_count = 1
				}
			}
			else = {
				clear_variable = election_count
			}
			# house calculations
			clear_variable = temp_total
			log = "[THIS.GetName]"
			for_each_loop = {
				array = electoral_pie_chart_number_array
				set_temp_variable = {
					temp_seats = electoral_pie_chart_number_array^i
				}
				# put a party popularity to a temp var
				divide_temp_variable = {
					temp_seats = 100
				}
				# divide party popularity so you get 0.65 from 65% for example
				multiply_temp_variable = {
					temp_seats = electoral_vote
				}
				# multiply that by number of seats to get seat share
				round_temp_variable = temp_seats				# round it up
				set_variable = {
					electoral_seat_number_array^i = temp_seats
				}
				# set electoral seat to new value
				add_to_variable = {
					temp_total = temp_seats
				}
				log = "[?i] - [?electoral_seat_number_array^i] ([?temp_total])"
			}
			# extra to make sure it adds up to electoral seats in state
			if = {
				limit = {
					check_variable = {
						temp_total < electoral_vote
					}
					# if total is less than number of seats
				}
				set_temp_variable = {
					temp = electoral_vote
				}
				# put number of seats to a temp var
				subtract_from_temp_variable = {
					temp = temp_total
				}
				# get seat difference
				find_highest_in_array = {
					array = electoral_seat_number_array
					index = highest_index
				}
				add_to_variable = {
					electoral_seat_number_array^highest_index = temp
				}
				# add seat difference to most popular party
			}
			else_if = {
				limit = {
					check_variable = {
						temp_total > electoral_vote
					}
					# if total is more than number of seats
				}
				subtract_from_variable = {
					temp_total = electoral_vote
				}
				# get seat difference
				find_highest_in_array = {
					array = electoral_seat_number_array
					index = highest_index
				}
				subtract_from_variable = {
					electoral_seat_number_array^highest_index = temp_total
				}
				# subtract seat difference from most popular party
			}
		}
	}
}

TEX_add_electoral_popularity = {
	set_temp_variable = {
		v = pop_type_added
	}
	custom_effect_tooltip = TEX_add_electoral_popularity_tt
	set_temp_variable = {
		temp = electoral_pie_chart_number_array^pop_type_added
	}
	add_to_temp_variable = {
		temp = pop_value_added
	}
	if = {
		limit = {
			check_variable = {
				temp > 100
			}
		}
		subtract_from_temp_variable = {
			temp = 100
		}
		subtract_from_temp_variable = {
			pop_value_added = temp
		}
	}
	if = {
		limit = {
			check_variable = {
				temp < 0
			}
		}
		multiply_temp_variable = {
			temp = -1
		}
		subtract_from_temp_variable = {
			pop_value_added = temp
		}
	}
	for_each_loop = {
		array = electoral_pie_chart_number_array
		add_to_array = {
			percentage_popularity = electoral_pie_chart_number_array^i
		}
	}
	add_percentage_popularity = yes
	for_each_loop = {
		array = percentage_popularity
		set_variable = {
			electoral_pie_chart_number_array^i = percentage_popularity^i
		}
	}
	hidden_effect = {
		TEX_stop_liberalism_from_state = yes
		TEX_ensure_one_hundred_percent = yes
		TEX_Election_Update_Lean = yes
	}
	clear_array = percentage_popularity
	TEX_create_pie_chart = yes
	update_texas = yes
}

	if = {

				check_variable = {
					TEX_er_who = 2
				}
				check_variable = {
					TEX_er_who = 3
				}
			}
		}
		set_temp_variable = {
			TEX_er_op1 = 0
		}
		set_temp_variable = {
			TEX_er_op2 = 1
		}
	}
	random_owned_state = {
		limit = {
			TEX_get_primary_states = yes
		}
		set_temp_variable = {
			pop_type_added = TEX.TEX_er_who
		}
		# 0 is auth dem, 1 is conservative etc. refer to order of ids
		set_temp_variable = {
			pop_value_added = 2
		}
		# percent added
		TEX_add_electoral_popularity = yes
		hidden_effect = {
			##Get all of the opposition
			set_temp_variable = {
				pop_opposition_tot = electoral_pie_chart_number_array^TEX_er_op1
			}
			#Get Opposition 1's Party Popularity in the State
			add_to_temp_variable = {
				pop_opposition_tot = electoral_pie_chart_number_array^TEX_er_op2
			}
			#Get Opposition 2's Party Popularity in the State
			round_temp_variable = pop_opposition_tot			#Just in case.
			#Divide opposition one with total.
			set_temp_variable = {
				pop_opposition_one = electoral_pie_chart_number_array^TEX_er_op1
			}
			divide_temp_variable = {
				pop_opposition_one = pop_opposition_tot
			}
			#This returns it as 0.XXXXXXXXX
			#Round in Hearts of Iron IV is Integer, can't handle Float. It sucks.
			multiply_temp_variable = {
				pop_opposition_one = 100
			}
			#Multiply 0.XXXXXXX to XX.XXXXXXXX
			round_temp_variable = pop_opposition_one			#Make it XX
			divide_temp_variable = {
				pop_opposition_one = 100
			}
			#Divided it to 0.XX
			#Divide opposition two with total.
			set_temp_variable = {
				pop_opposition_two = electoral_pie_chart_number_array^TEX_er_op2
			}
			divide_temp_variable = {
				pop_opposition_two = pop_opposition_tot
			}
			#Round in Hearts of Iron IV is Integer, can't handle Float. It sucks.
			multiply_temp_variable = {
				pop_opposition_two = 100
			}
			#Multiply 0.XXXXXXX to XX.XXXXXXXX (ex. 0.565217391304 => 56.5217391304
			round_temp_variable = pop_opposition_two			#Make it XX (ex. 56)
			divide_temp_variable = {
				pop_opposition_two = 100
			}
			#Divided it to 0.XX (0.56)
			#Right, time to determine how much parts each opposition should get.
			IF = {
				limit = {
					#Check if neither is above 85%.
					NOT = {
						check_variable = {
							pop_opposition_one > 0.85
						}
						check_variable = {
							op_opposition_two > 0.85
						}
					}
				}
				set_temp_variable = {
					pop_opposition_one_deduct = 2
				}
				multiply_temp_variable = {
					pop_opposition_one_deduct = pop_opposition_one
				}
				set_temp_variable = {
					pop_opposition_two_deduct = 2
				}
				multiply_temp_variable = {
					pop_opposition_two_deduct = pop_opposition_two
				}
			}
			else_if = {
				limit = {
					check_variable = {
						pop_opposition_one > 0.85
					}
				}
				set_temp_variable = {
					pop_opposition_one_deduct = 2
				}
				set_temp_variable = {
					pop_opposition_two_deduct = 0
				}
			}
			else_if = {
				limit = {
					check_variable = {
						op_opposition_two > 0.85
					}
				}
				set_temp_variable = {
					pop_opposition_one_deduct = 0
				}
				set_temp_variable = {
					pop_opposition_two_deduct = 2
				}
			}
			set_temp_variable = {
				pop_type_subtracted = TEX.TEX_er_op1
			}
			# 0 is auth dem, 1 is conservative etc. refer to order of ids
			set_temp_variable = {
				pop_value_subtracted = pop_opposition_one_deduct
			}
			# percent added
			TEX_subtract_electoral_popularity = yes
			set_temp_variable = {
				pop_type_subtracted = TEX.TEX_er_op2
			}
			# 0 is auth dem, 1 is conservative etc. refer to order of ids
			set_temp_variable = {
				pop_value_subtracted = pop_opposition_two_deduct
			}
			# percent added
			TEX_subtract_electoral_popularity = yes
		}
	}

			subtract_from_temp_variable = {
				attack_power = 3
			}
			custom_effect_tooltip = TEX_ap_decrease_by_3_tt
		}
		if = {
			limit = {
				TEX = {
					has_completed_focus = TEX_focus_on_reform
				}
				OR = {
					AND = {
						#SocDem
						check_variable = {
							governor = 3
						}
						TEX = {
							check_variable = {
								primary_candidate = 2
							}
						}
					}
					AND = {
						#AuthDem
						check_variable = {
							governor = 0
						}
						TEX = {
							check_variable = {
								primary_candidate = 0
							}
						}
					}
				}
			}
			add_to_temp_variable = {
				attack_power = 3
			}
			custom_effect_tooltip = TEX_ap_increase_by_3_tt
		}
		#Remove from the victim
		set_temp_variable = {
			pop_type_subtracted = TEX.TEX_ea_victim
		}
		# 0 is auth dem, 1 is conservative etc. refer to order of ids
		set_temp_variable = {
			pop_value_subtracted = attack_power
		}
		# percent added
		TEX_subtract_electoral_popularity = yes
		#Add to the attacker
		set_temp_variable = {
			pop_type_added = TEX.TEX_ea_attacker
		}
		# 0 is auth dem, 1 is conservative etc. refer to order of ids
		set_temp_variable = {
			pop_value_added = attack_power
		}
		# percent added
		TEX_add_electoral_popularity = yes
	}
