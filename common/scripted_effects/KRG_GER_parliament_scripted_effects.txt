GER_Start_Law_Process = {
	if = {
		limit = {
			has_country_flag = GER_der_neue_staat
		}		
		meta_effect = {
			text = {
				GER_law_[LAW_ID]_effect = yes
			}
			LAW_ID = "[?GER_active_law]"
		}
	}
	else = {
		set_country_flag = GER_law_introduced_Reichstag
		set_country_flag = GER_voting_on_law
		set_variable = { GER_active_law = GER_active_law}
		
		meta_effect = {
			text = {
				GER_law_[LAW_ID]_start = yes
			}
			LAW_ID = "[?GER_active_law]"
		}
	
		activate_mission = GER_voting_law_in_reichstag
	}
}

	clear_variable = bundesrat_support_value

	for_each_loop = {
		array = bundesrat_member_array

		if = {
			limit = {
				check_variable = { v:bundesrat_support > 1 }
			}
			add_to_variable = { bundesrat_support_value = v:bundesrat_seat }
		}
	}
}

# main code for adding/subtracting seats from bundesrat member
modify_bundesrat_seat = {
	set_temp_variable = { v = member }
	custom_effect_tooltip = GER_modify_bundesrat_seat_tt

	if = {
		limit = {
			check_variable = { member > 0 }
		}
		var:member = {
			add_to_variable = { bundesrat_seat = seat_value }
		}
	}

	add_to_variable = { bundesrat_total_members = seat_value }

	set_temp_variable = { temp = bundesrat_total_members }
	modulo_temp_variable =  { temp = 2 }
	set_variable = { bundesrat_support_majority = bundesrat_total_members }
	divide_variable = { bundesrat_support_majority = 2 }
	add_to_variable = { bundesrat_support_majority = temp }

	update_bundesrat_support = yes
}

modify_reichstag_seat = {
	set_temp_variable = { v = member }
	custom_effect_tooltip = GER_modify_reichstag_seat_tt

	if = {
		limit = {
			check_variable = { member > 0 }
		}

		for_loop_effect = {
			start = 1
			end = reichstag_member_array^num
			compare = less_than_or_equals

			add_to_temp_array = { temp_array = v }
		}

		remove_from_temp_array = { array = temp_array value = member }

		if = {
			limit = {
				set_temp_variable = { temp_total = member:reichstag_seat }
				add_to_temp_variable = { temp_total = seat_value }
				check_variable = { temp_total < 0 }
			}
			set_temp_variable = { seat_value = member:reichstag_seat }
			multiply_temp_variable = { seat_value = -1 }
		}
		else_if = {
			limit = {
				check_variable = { temp_total > reichstag_main_array^num }
			}
			set_temp_variable = { corrected_value = temp_total }
			subtract_from_temp_variable = { corrected_value = reichstag_main_array^num }
			subtract_from_temp_variable = { seat_value = corrected_value }
		}

		add_to_variable = { member:reichstag_seat = seat_value }

		while_loop_effect = {
			limit = {
				OR = {
					check_variable = { seat_value > 0 }
					check_variable = { seat_value < 0 }
				}
			}

			for_each_loop = {
				array = temp_array

				if = {
					limit = {
						check_variable = { seat_value = 0 }
					}
					set_temp_variable = { break = 1 }
				}
				else_if = {
					limit = {
						check_variable = { v:reichstag_seat > 0 }
						check_variable = { seat_value > 0 }
					}
					subtract_from_variable = { v:reichstag_seat = 1 }
					subtract_from_temp_variable = { seat_value = 1 }
				}
				else_if = {
					limit = {
						check_variable = { v:reichstag_seat > 0 }
						check_variable = { seat_value < 0 }
					}
					add_to_variable = { v:reichstag_seat = 1 }
					add_to_temp_variable = { seat_value = 1 }
				}
			}
		}
	}
}

# main code for unsorted version of adding seat
add_reichstag_seat = {
	for_loop_effect = {
		start = 0
		end = reichstag_seat_array^num

		add_to_temp_array = { temp_reichstag_seat_array = v }
	}

	subtract_from_temp_variable = { member_added = 1 }
	remove_from_temp_array = { array = temp_reichstag_seat_array value = member_added }
	add_to_variable = { reichstag_seat_array^member_added = seat_value_added }

	while_loop_effect = {
		limit = { check_variable = { seat_value_added > 0 } }

		for_loop_effect = {
			start = 0
			end = temp_reichstag_seat_array^num

			if = {
				limit = {
					check_variable = { reichstag_seat_array^temp_reichstag_seat_array^v > 0 }
					check_variable = { seat_value_added > 0 }
				}
				subtract_from_variable = { reichstag_seat_array^temp_reichstag_seat_array^v = 1 }
				subtract_from_temp_variable = { seat_value_added = 1 }
			}
		}
	}

	modify_reichstag_array = yes
}

# main code for unsorted version of subtracting seat
subtract_reichstag_seat = {
	for_loop_effect = {
		start = 0
		end = reichstag_seat_array^num

		add_to_temp_array = { temp_reichstag_seat_array = v }
	}

	subtract_from_temp_variable = { member_subtracted = 1 }
	remove_from_temp_array = { array = temp_reichstag_seat_array value = member_subtracted }
	subtract_from_variable = { reichstag_seat_array^member_subtracted = seat_value_subtracted }

	while_loop_effect = {
		limit = { check_variable = { seat_value_subtracted > 0 } }

		for_loop_effect = {
			start = 0
			end = temp_reichstag_seat_array^num

			if = {
				limit = {
					check_variable = { reichstag_seat_array^temp_reichstag_seat_array^v < 415 }
					check_variable = { seat_value_subtracted > 0 }
				}
				add_to_variable = { reichstag_seat_array^temp_reichstag_seat_array^v = 1 }
				subtract_from_temp_variable = { seat_value_subtracted = 1 }
			}
		}
	}

	modify_reichstag_array = yes
}
