reichspakt_aid_MOR = {
	allowed = {
		#Normal stuff
		OR = {
			tag = GER
			is_in_faction_with = GER
			check_variable = {
				reichspakt_alignment_level > 0
			}
		}
	}
	enable = {
		#Check if you're able to
		OR = {
			is_in_faction_with = MOR
			AND = {
				ROOT = {
					check_variable = {
						reichspakt_alignment_level > 0
					}
				}
				MOR = {
					check_variable = {
						reichspakt_alignment_level > 0
					}
				}
			}
		}
		#Morocco war
		MOR = {
			has_war_with = AMR
		}
		#If you've got war, don't aid MOR
		OR = {
			AND = {
				has_war = yes
				has_war_with = AMR
			}
			has_war = no
		}
		#Is Major or MAF but not the Dutch to keep it balanced (or else you'll have 20 reichspakt volunteers show up to the supply hellhole)
		NOT = {
			tag = HOL
		}
		OR = {
			has_country_flag = reichspakt_military_major_ai
			tag = MAF
		}
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = protect
		id = "MOR"
		value = 200
	}
	ai_strategy = {
		type = support
		id = "MOR"
		value = 100
	}
	ai_strategy = {
		type = send_volunteers_desire
		id = "MOR"
		value = 100
	}
}

reichspakt_contain_AMR = {
	allowed = {
		OR = {
			tag = GER
			is_in_faction_with = GER
			check_variable = {
				reichspakt_alignment_level > 0
			}
		}
		NOT = {
			OR = {
				tag = MOR
			}
		}
	}
	enable = {
		#Check if you're able to
		OR = {
			is_in_faction_with = MOR
			AND = {
				ROOT = {
					check_variable = {
						reichspakt_alignment_level > 0
					}
				}
				MOR = {
					check_variable = {
						reichspakt_alignment_level > 0
					}
				}
			}
		}
		#Morocco war
		OR = {
			AMR = {
				has_war_with = MOR
			}
			SPR = {
				has_war_with = MOR
			}
		}
		#If you've got war, don't contain AMR (unless you're literally at war with AMR)
		OR = {
			AND = {
				has_war = yes
				has_war_with = AMR
			}
			has_war = no
		}
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = contain
		id = "AMR"
		value = 200
	}
}
